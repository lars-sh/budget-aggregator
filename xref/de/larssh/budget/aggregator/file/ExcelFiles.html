76  	public static List<Budget> read(final Path source) throws IOException, StringParseException {
77  		final List<Budget> budgets = new ArrayList<>();
78  		try (
79  			InputStream inputStream = Files.newInputStream(source);
80  			Workbook workbook = WorkbookFactory.create(inputStream)) {
81  			final int numberOfSheets = workbook.getNumberOfSheets();
82  			for (int sheetIndex = 0; sheetIndex < numberOfSheets; sheetIndex += 1) {
83  				final Sheet sheet = workbook.getSheetAt(sheetIndex);
84  				final Set<Budget> sheetBudgets = Budget.of(readSheet(sheet));
85  				Budgets.setReference(sheetBudgets, BudgetReference.FILE_NAME, Nullables.orElseThrow(source.getFileName()).toString());
86  				Budgets.setReference(sheetBudgets, BudgetReference.SHEET, sheet.getSheetName());
87  				budgets.addAll(sheetBudgets);
88  			}
89  		}
90  		return budgets;
91  	}

95  		final int lastRowIndex = sheet.getLastRowNum();
96  		final int firstRowIndex = sheet.getFirstRowNum();
97  		final List<List<String>> data = new ArrayList<>(lastRowIndex - firstRowIndex + 1);
98  		for (int rowIndex = firstRowIndex; rowIndex <= lastRowIndex; rowIndex += 1) {
99  			data.add(readRow(sheet.getRow(rowIndex)));
100 		}
101 		return new Csv(data);

287 		private String getBudgetColumnName(final Budget budget) {
288 			// 1. Try to find a cached name
289 			final String cachedName = budgetColumnNames.get(budget);
290 			if (cachedName != null) {
291 				return cachedName;
292 			}
293 			// 2. Create the straight-forward column name
294 			final String simpleName = String.format("%s %d", budget.getType().getName(), budget.getYear());
295 			String columnName = simpleName;
296 			// 3. Add a number in case of multiple columns with the same name
297 			for (int count = 2; budgetColumnNamesUsed.contains(columnName); count += 1) {
298 				columnName = String.format("%s (%d)", simpleName, count);
299 			}
300 			// 4. Add to cache and return
301 			budgetColumnNamesUsed.add(columnName);
302 			budgetColumnNames.put(budget, columnName);
303 			return columnName;
304 		}

306 		private Optional<CellStyle> getCellStyle(final Workbook workbook, final SimplifiedCellStyle simplifiedCellStyle, final Optional<String> dataFormat) {
307 			if (simplifiedCellStyle == SimplifiedCellStyle.NORMAL && !dataFormat.isPresent()) {
308 				return Optional.empty();
309 			}
310 			return Optional.of(cellStyleCache.computeIfAbsent(simplifiedCellStyle.name() + ';' + dataFormat.orElse(""), key -> {
311 				final CellStyle cellStyle = workbook.createCellStyle();
312 				dataFormat.ifPresent(format -> cellStyle.setDataFormat(workbook.createDataFormat().getFormat(format)));
313 				if (simplifiedCellStyle != SimplifiedCellStyle.NORMAL) {
314 					final Font font = workbook.createFont();
315 					if (simplifiedCellStyle == SimplifiedCellStyle.BOLD) {
316 						font.setBold(true);
317 					} else if (simplifiedCellStyle == SimplifiedCellStyle.ITALIC) {
318 						font.setItalic(true);
319 					}
320 					cellStyle.setFont(font);
321 				}
322 				return cellStyle;
323 			}));
324 		}

326 		@PackagePrivate
327 		void write() throws IOException {
328 			try (XSSFWorkbook workbook = new XSSFWorkbook()) {
329 				workbook.setCellFormulaValidation(false);
330 				writeProducts(workbook.createSheet(SHEET_NAME_BALANCES));
331 				writeAccounts(workbook.createSheet(SHEET_NAME_ACCOUNTS));
332 				// Auto Size Columns
333 				workbook.sheetIterator().forEachRemaining(sheet -> {
334 					final int numberOfColumns = sheet.getRow(0).getLastCellNum();
335 					for (int columnIndex = 0; columnIndex < numberOfColumns; columnIndex += 1) {
336 						sheet.autoSizeColumn(columnIndex);
337 						// Add the width of the auto filter
338 						final double widthOfHeader = SheetUtil.getColumnWidth(sheet, columnIndex, false, 0, 0);
339 						if (widthOfHeader != -1) {
340 							final int intWidth = (int) Math.round(Math.min(CHARACTER_WIDTH * widthOfHeader + AUTO_FILTER_WIDTH, COLUMN_MAX_WIDTH));
341 							if (intWidth > sheet.getColumnWidth(columnIndex)) {
342 								sheet.setColumnWidth(columnIndex, intWidth);
343 							}
344 						}
345 					}
346 				});
347 				workbook.write(outputStream);
348 			}
349 		}
