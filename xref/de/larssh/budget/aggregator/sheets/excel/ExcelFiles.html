130 			final StringBuilder comment = new StringBuilder();
131 			for (final Entry<BudgetReference, String> entry : budget.getReferences().entrySet()) {
132 				if (comment.length() > 0) {

264 		private String getBudgetColumnName(final Budget budget) {
265 			// 1. Try to find a cached name
266 			final String cachedName = budgetColumnNames.get(budget);
267 			if (cachedName != null) {
268 				return cachedName;
269 			}
270 			// 2. Create the straight-forward column name
271 			final String simpleName = String.format("%s %d", budget.getType().getName(), budget.getYear());
272 			String columnName = simpleName;
273 			// 3. Add a number in case of multiple columns with the same name
274 			for (int count = 2; budgetColumnNamesUsed.contains(columnName); count += 1) {
275 				columnName = String.format("%s (%d)", simpleName, count);
276 			}
277 			// 4. Add to cache and return
278 			budgetColumnNamesUsed.add(columnName);
279 			budgetColumnNames.put(budget, columnName);
280 			return columnName;
281 		}

283 		private Optional<CellStyle> getCellStyle(final Workbook workbook, final SimplifiedCellStyle simplifiedCellStyle, final Optional<String> dataFormat) {
284 			if (simplifiedCellStyle == SimplifiedCellStyle.NORMAL && !dataFormat.isPresent()) {
285 				return Optional.empty();
286 			}
287 			return Optional.of(cellStyleCache.computeIfAbsent(simplifiedCellStyle.name() + ';' + dataFormat.orElse(""), key -> {
288 				final CellStyle cellStyle = workbook.createCellStyle();
289 				dataFormat.ifPresent(format -> cellStyle.setDataFormat(workbook.createDataFormat().getFormat(format)));
290 				if (simplifiedCellStyle != SimplifiedCellStyle.NORMAL) {
291 					final Font font = workbook.createFont();
292 					if (simplifiedCellStyle == SimplifiedCellStyle.BOLD) {
293 						font.setBold(true);
294 					} else if (simplifiedCellStyle == SimplifiedCellStyle.ITALIC) {
295 						font.setItalic(true);
296 					}
297 					cellStyle.setFont(font);
298 				}
299 				return cellStyle;
300 			}));
301 		}

303 		@PackagePrivate
304 		void write() throws IOException {
305 			try (XSSFWorkbook workbook = new XSSFWorkbook()) {
306 				workbook.setCellFormulaValidation(false);
307 				writeProducts(workbook.createSheet(SHEET_NAME_PRODUCTS));
308 				writeAccounts(workbook.createSheet(SHEET_NAME_ACCOUNTS));
309 				// Auto Size Columns
310 				workbook.sheetIterator().forEachRemaining(sheet -> {
311 					final int numberOfColumns = sheet.getRow(0).getLastCellNum();
312 					for (int columnIndex = 0; columnIndex < numberOfColumns; columnIndex += 1) {
313 						sheet.autoSizeColumn(columnIndex);
314 						// Add the width of the auto filter
315 						final double widthOfHeader = SheetUtil.getColumnWidth(sheet, columnIndex, false, 0, 0);
316 						if (widthOfHeader != -1) {
317 							final int intWidth = (int) Math.round(Math.min(CHARACTER_WIDTH * widthOfHeader + AUTO_FILTER_WIDTH, COLUMN_MAX_WIDTH));
318 							if (intWidth > sheet.getColumnWidth(columnIndex)) {
319 								sheet.setColumnWidth(columnIndex, intWidth);
320 							}
321 						}
322 					}
323 				});
324 				workbook.write(outputStream);
325 			}
326 		}

354 		private void appendProductBudgets(final XSSFSheet sheet, final Set<Product> products) {
355 			for (final Budget budget : budgets) {
356 				// Header
357 				final Cell headerCell = appendString(sheet.getRow(0), getSimplifiedCellStyle(budget), getBudgetColumnName(budget));
358 				addBudgetsComment(headerCell, budget);
359 				// Balances
360 				int rowIndex = 1;
361 				for (int i = products.size(); i > 0; i -= 1) {
362 					appendFormula(sheet.getRow(rowIndex), getSimplifiedCellStyle(budget), DATA_FORMAT_CURRENCY, String.format("SUMIFS(%1$s[%7$s], %1$s[%3$s], %2$s[[#This Row],[%3$s]], %1$s[%4$s], %2$s[[#This Row],[%5$s]], %1$s[%6$s], TRUE)", SHEET_NAME_ACCOUNTS,  // 1
363 					SHEET_NAME_PRODUCTS,  // 2
364 					COLUMN_NAME_PRODUCT,  // 3
365 					COLUMN_NAME_PRODUCT_DESCRIPTION,  // 4
366 					COLUMN_NAME_DESCRIPTION,  // 5
367 					COLUMN_NAME_SUM,  // 6
368 					getBudgetColumnName(budget))); // 7
369 					rowIndex += 1;
370 				}
